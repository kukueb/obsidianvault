#### Создание однонаправленного списка из N элементов. Список строится в обратном порядке - стековом - добавлением элемента в начало списка.
```c++
#include <iostream>

using namespace std;

struct list {
  int info;
  list *next;
};

list *make_list(int n) {
  list *beg = new (list);
  cin >> beg->info;
  beg->next = NULL;
  list *r;
  for (int i = 1; i < n; ++i) {
    r = new (list);
    cin >> r->info;
    r->next = beg;
    beg = r;
  }
  return beg;
}
```

Удаление списка
```c++
p = beg;
while (p != NULL) {
	list *r = p;
	p = p->next;
	delete r;
}
```

*Рекурсивное удаление списка*
```c++
void Delete_List(list* beg) {
	if (beg != nullptr) {
		Delete_List(beg->next);
		delete beg;
	}
}
```

*Вывод*
```c++
void print(list* beg) {
	list *p = beg;

}
```
