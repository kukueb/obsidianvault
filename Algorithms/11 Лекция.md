Задача проверки правильности скобочной последовательности
```cpp
#include <iostream>
#include <stack>

using namespace std;

int main() {
  stack<char> st;
  bool        flag = true;
  for (char x = getchar(); x != '\n'; x = getchar()) {
    // считываем посимвольно до Enter
    switch (x) {
    case '(':
    case '[': {
      st.push(x);
      break;
    }
    case ')': {
      flag = flag && (st.top() == '(');
      st.pop();
      break;
    }
    case ']': {
      flag = flag && (st.top() == '(');
      st.pop();
      break;
    }
    default:
      flag = false; // введён неверный символ
    }
  }
  flag = flag && st.empty(); // проверка не остались ли ещё открытые скобки
  if (flag)
    cout << "Да" << endl;
  else
    cout << "Нет" << endl;
  return 0;
}
```

# Контейнер очереди `queue`
```cpp
#include <queue>

queue < <тип_элемента> > <имя_очереди>;
queue <int> q;
```

**Операции**:
- `push()`
- `pop()`
- `front()` - получить элемент в начале очереди, сам элемент остаётся в очереди
- `back()` - аналогично `front()`
- `size()`
- `empty()`

## Очередь с приоритетом (`priority queue`)
Очередь с приоритетом - это очередь, в которой новый элемент добавляется так, чтобы очередь была отсортирована по убыванию.
*Очередь с приоритетом больше похожа на стек, чем на очередь*
Для добавления элементов используется функция `push()`, а для обращения к первому элементу - `top()`.
Функция `back()` для очереди с приоритетом **не реализована**

**Пример**
```cpp
#include <iostream>
#include <queue>

using namespace std;

int main() {
  priority_queue<string> queue;
  queue.push("Tom");
  queue.push("Bob");
  queue.push("Sam");

  cout << "queue size: " << queue.size() << endl;
  cout << "First: " << queue.top() << endl;
  return 0;
}
```

Найти кол-во нечётных чисел
```cpp
#include <iostream>
#include <queue>

using namespace std;

struct Tree {
  int   info;
  Tree *left;
  Tree *right;
};

// ... стандартные методы Tree

int BFScount(Tree *root) {
  if (root == NULL)
    return 0;
  int           counter = 0;
  queue<Tree *> q;
  q.push(root);
  while (!q.empty()) {
    Tree *node = q.front();
    q.pop();
    if (node->left != NULL)
      q.push(node->left);
    if (node->right != NULL)
      q.push(node->right);
    if ((node->info & 1) == 0) // проверяем чётность
      ++counter;
  }
  return counter;
}
```

# `Deque`
```cpp
deque <int> d;
```
**Методы**:
- `front()`
- `back()`
- `push_front()`
- `push_back()`
- `pop_front()`
- `pop_back()`
- `size()`
- `clear()`

`Deque` является **двунаправленной очередью** *(`list` однонаправленный)*

**Пример**. Определить является ли последовательность симметричной

```cpp
#include <iostream>
#include <deque>

using namespace std;

int main() {
	int x;
	deque <int> d;
	for (cin >> x; x != 0; cin >> x) {
		d.push_back(x);
	}

	while (!d.empty() && d.front() == d.back()) {
		d.pop_front();
		if (!d.empty())
			d.pop_back();
	}
	
	cout << endl << (d.empty() ? "Да" : "Нет") << endl << endl;
	return 0;
}

```

# Ассоциативные контейнеры
## `set`
Контейнер `set` описывает множество, состоящее из уникальных значений
```cpp
#include <set>
set <type> <имя>;
```
Методы `set`:
- `empty()`
- `size()`
- `max_size()` - максимальное возможное кол-во элементов во множестве
- `clear()` - очищает множество
- `insert()` - добавить элемент
- `erase()` - позволяет удалить элементы;
- `swap()` - обменивает содержимое двух множеств
- `count()` - кол-во элементов, соответствующих определённому ключу
- `find()` - находит элемент с конкретным ключом. Возвращает константный итератор позиции элемента или итератор `end()`, если таковой не найден
- `lower_bound()` - возвращает итератор на первый элемент, который меньше, чем определённый ключ
- `upper_bound()` - возвращает итератор на первый элемент, который больше, чем определённый ключ

```cpp
#include <iostream>
#include <set>

using namespace std;

int main(int argc, char *argv[]) {
  set<int> numbers{3, 4, 5};
  numbers.insert(1);
  numbers.insert(2);
  numbers.insert(2);
  numbers.insert(2);
  numbers.insert(6);

  for (int n : numbers) {
    cout << n << "\n";
  }
  cout << endl;

  cout << "count 2 = " << numbers.count(2) << endl; // 3 / для g++ не работает?
  return 0;
}
```

```cpp
set<int> numbers{2,3,4,5};
numbers.erase(1);
numbers.erase(2);
numbers.erase(5);
for (int n : numbers) {
	cout << n << endl;
}
```

Функция `count()` проверяет наличие значения во множестве. Если определённое значение имеется, то функция возвращает 1, если нет - то 0
