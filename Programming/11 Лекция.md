### Перегрузка функций
Существует механизм перегрузки, который позволяет изменять типы фактических параметров
```c++
double frac(int a, int b) {
	return (double)a/b;
}
double frac(double a, double b) {
	return a/b;
}

...

cout << frac(7, 3) << endl; // работает первая функция
cout << frac(7.5, 2.5) << endl; // работает вторая функция

cout << frac(3.14, 2) << endl; // ошибка компиляции
```

**Пример** (параметр по умолчанию):
```c++
double frac(int a, int b, double c = 3.14) {
	return (double)(a/b)*c;
}
```

### Массивы в параметрах функции
Если в функцию передаётся массив, то даже если это организовано, как передача по значению, автоматически передаётся **указатель (адрес) на первый элемент массива**  
Следующие объявления функции равноценны:
```c++
void print(int numbers[]);
void print(int *numbers);
```

```c++
void print(int[]);

int main() {
	int nums[] {1, 2, 3, 4, 5};
	print(nums);
}

void print(int numbers[]) {
	cout << "First number: " << numbers[0] << endl; // 1
}
```

```c++
void print(int[]);

int main() {
	int nums[] {1, 2, 3, 4, 5};
	print(nums);
}

void print(int numbers[]) {
	cout << "First number: " << *numbers << endl; // 1
}
```

```c++
int f(int *a) // *a = 100
```

#### Ограничения
При передаче массива в функцию передаётся только адрес первого элемента.
Для того, чтобы корректно обработать весь массив, необходимо знать его длину либо с помощью специальной функции, либо передавая длину явно.
```c++
void print(int numbers[]) {
	int size = sizeof(numbers) / sizeof(numbers[0]);
	// или так size_t size = size(nums);
	cout << size << endl;
}
```

```c++
void print(int[], size_t);

int main() {
	int nums[] {1,2,3,4,5};
	size_t n {size(nums)};
	print(nums, n);
}

void print(int numbers[], size_t n) {
	for (size_t i {}; i < n; i++) {
		cout << numbers[i] << "\t";
	}
}
```

### Константные массивы
Для запрета изменения *элементов массива*, внутри функции можно использовать константный массив (префикс `const`)

```c++
void print(const int* numbers, const size_t n) {
	for (size_t i {}; i < n; ++i) {
		cout << numbers[i] << "\t";
	}
	cout << endl;
}
void twice(int* numbers, size_t n) {
	for (size_t i {}; i < n; ++i) {
		numbers[i] = numbers[i]*2;
	}
}

int main() {

	int nums[] {1,2,3,4,5};
	print(nums); // 1 2 3 4 5
	twice(nums);
	print(nums); // 2 4 6 8 10
}

```

### Передача многомерного массива
При передаче многомерного массива, передаётся адрес на первую строку - массив
```c++
void print(const int(*)[3], const int);

int main() {
	int table[][3] { {1,2,3}, {4,5,6}, {7,8,9} };
	// количество строк или подмассивов
	print(table, 3);
}

void print(const int (*rows)[3], const int n) {
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			cout << rows[i][j] << "\t";
		}	
		cout << endl;
	}
}
```

### Рекурсия
Если функция вызывает сама себя, это называется прямой рекурсией.
Если функция **A** вызывает функцию **B**, а функция **B** функцию **A**, такая рекурсия называется косвенной.
Необходимым, но недостаточным условием рекурсивной функции, является:
1) наличие хотя бы одной не рекурсивной ветки
2) изменение параметра рекурсии
**Пример 1**
```c++
int fib(int n) {
	if (n == 1 || n == 2) return 1;
	else return fib(n-1) + fib(n-2);
}

int main(void) {
	cout << fib(10) << endl;
	return 0;
}
```

**Пример 2**
Возведение в степень
```c++
int power(long int x, unsigned int y) {
	int d = 0;
	if (y == 0) return 1;
	else return x * power(x, y-1);
}
```

**Пример 3**
```c++
int sum(int n) {
	if (n == 1) return 1;
	else return sum(n-1) + n;
}

int main() {
	cout << "1..5: " << sum(5) << endl;
}
```

**Пример 4**
Дан массив. Найти сумму
```c++
int sum(int i) {
	if (i == n) return 0;
	else if (a[i] > 0) return sum(i + 1) + a[i];
	else return sum(i + 1);
}
```

### Косвенная рекурсия
Алгоритм вычисления значений функций F(w) и Q(w), где w - натуральное число, задан следующими соотношениями: ...
Реализация
```c++
int F(int n) {
	if (n == 1) return 1;
	else return F(n-1) + 2*Q(n-1);
}

int Q(int n) {
	if (n == 1) return 1;
	else return Q(n-1) - 2*F(n-1);
}
```
